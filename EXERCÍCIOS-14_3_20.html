# 21) Crie uma fun√ß√£o para resolver equa√ß√µes de segundo grau, onde: se delta<0, teremos ‚Äúra√≠zes
# complexas‚Äù, ou ent√£o, teremos raiz1 e raiz2, lembrando que para calcular as ra√≠zes, o delta √© b^2-
#   4*a*c. Teste a fun√ß√£o criada para:
equacao2grau <- function(a, b, c)
{
  delta <- b^2-4*a*c
  if (delta<0)
    print("ra√≠zes complexas")
  else {
    raiz1 <- (-(b)+sqrt(delta))/(2*a)
    raiz2 <- (-(b)-sqrt(delta))/(2*a)
    print(raiz1)
    print(raiz2)
  }
}

## [1] -6
# b) x¬≤ + 3x + 5
